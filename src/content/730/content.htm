    import peasy.*;
    PeasyCam cam;
    float steps = 2000;
      
    void setup() {
      size(1500, 1500, P3D);
      cam = new PeasyCam(this, 6000);
      background(255);
      stroke(0);
      //fill(0);
      noFill();
      frameRate(60);
      smooth();
    } 
    
    
    void draw() {
     
      float osc = random(10);
     // float osc = 1;
      
      for  (int i = 1; i < steps; i++) {
          rotate(730);
          point( (width/steps) * i , (width/steps) * i + osc, osc * 2);
        osc += random(-10, 11);
        rotate(2.2);
      //  osc += 0.01;
      }
      
    }
    
    void mouseClicked(){
       saveFrame(); 
      
      
    }
